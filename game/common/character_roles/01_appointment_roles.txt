# Roles that are assigned from code when a character gets an appointment

character_role_ig_leader = {
	texture = "gfx/interface/icons/character_role_icons/politician.dds"

	type = politician
	priority = 25
	
	on_career_end = {
		if = {
			limit = {
				has_variable = former_shogun_var # designates Taikun
				owner ?= {
					country_is_taikunate = yes
				}
			}
			owner ?= {
				debug_log = "[TimeKeeper.GetCurrentDate.GetString]:Taikun retired: new Taikun designated - [THIS.GetCountry.GetNameNoFormatting]"
			}
			character_designate_new_taikun = yes
		}
		remove_as_interest_group_leader = yes
	}
	
	character_modifier = {
		character_prominence_add = 25
	}
}
