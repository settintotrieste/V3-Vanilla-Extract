taikunate_liberal_party_name_sgui = {
    scope = country

    effect = {
        if = {
            limit = {
                py:liberal_party ?= {
                    party_has_members = yes
                }
            }
            py:liberal_party ?= {
                if = {
                    limit = {
                        any_member = {
                            leader ?= {
                                has_variable = former_shogun_var # designates Taikun
                            }
                        }
                    }
                    custom_tooltip_no_bullet = "TAIKUNATE_TAIKUN_PARTY_NAME"
                }
                else = {
                    ordered_member = {
                        order_by = ig_clout
                        position = 0
                        leader = {
                            save_temporary_scope_as = relevant_leader
                        }
                        if = {
                            limit = {
                                scope:relevant_leader ?= {
                                    has_variable = daimyo_var
                                }
                            }
                            custom_tooltip_no_bullet = "TAIKUNATE_LIBERAL_PARTY_DOMAIN_NAME"
                        }
                        else_if = {
                            limit = {
                                scope:relevant_leader ?= {
                                    is_noble = yes
                                }
                            }
                            custom_tooltip_no_bullet = "TAIKUNATE_LIBERAL_PARTY_LEADER_NAME"
                        }
                        else = {
                            custom_tooltip_no_bullet = "TAIKUNATE_LIBERAL_PARTY_STATE_NAME"
                        }
                    }
                }
            }
        }
        else = {
            custom_tooltip_no_bullet = "TAIKUNATE_LIBERAL_PARTY_NAME_FALLBACK" # set key
        }
    }
}
