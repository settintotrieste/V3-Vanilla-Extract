# Root = State Region
us_american_buildings_trigger = {
	is_in_geographic_region = geographic_region_north_america
	any_scope_state = {
		owner = {
			OR = {
				country_has_primary_culture = cu:yankee
				country_has_primary_culture = cu:dixie
				country_has_primary_culture = cu:afro_american
				country_has_primary_culture = cu:cajun
			}
		}
	}
}

latin_american_buildings_trigger = {
	OR = {
		is_in_geographic_region = geographic_region_iberia_old
		is_in_geographic_region = geographic_region_south_america
		is_in_geographic_region = geographic_region_central_america
		is_in_geographic_region = geographic_region_philippines
	}
}

french_buildings_trigger = {
	OR = {
		is_in_geographic_region = geographic_region_france_old
		is_in_geographic_region = geographic_region_occitania_old
		this = s:STATE_ALSACE_LORRAINE
		this = s:STATE_WALLONIA
		this = s:STATE_SAVOY
		this = s:STATE_CORSICA
		this = s:STATE_WEST_SWITZERLAND
		this = s:STATE_WALLONIA
	}
}
